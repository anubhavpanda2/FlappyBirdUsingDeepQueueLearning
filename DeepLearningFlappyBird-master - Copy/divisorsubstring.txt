#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int divisorsubstring(int n,int k)
{
	int d=pow(10,k);
	int res=0;
	int ncopy=n;
	unordered_set<int>s;
	int c;
	int cnt=0;
	auto st=to_string(n);
	 n=st.size();
	for(int i=0;i<n;i++)
	{
		c=st[i]-'0';
		res=res*10+c;
		cnt++;
		if(cnt>=k)
		{
			res=res%d;
		s.insert(res)	;
		}
		
	}
	int out=0;
	for(auto a:s)
	{
		if(ncopy%a==0)
		out++;
	}
	return out;
}
int main() {
	// your code goes here
	cout<<divisorsubstring(555,1)<<endl;
	cout<<divisorsubstring(120,2)<<endl;
	cout<<divisorsubstring(5341,2)<<endl;
	return 0;
}
